Peamble for nice formatting

install.packages("../eider/", repos = NULL, type = "source")
library(eider)
# Formatting
library(knitr)
library(kableExtra)
library(tidyverse)
kbl <- function(x) {
  x <- head(x, 10)
  x %>%
    knitr::kable(format = 'html', 
                 table.attr = 'data-quarto-disable-processing="true"') %>%
    kableExtra::kable_styling(bootstrap_options = "striped", 
                              full_width = FALSE)
}
